CREATE TABLE "atacmonitor_keyspace"."stops"(
	"stop_id" ascii,
	"route_id" int,
	"timestamp" timestamp,
	"waiting_time" int,
	PRIMARY KEY(("stop_id", "route_id"), "timestamp"))
WITH CUSTOM_PROPERTIES = {
	'capacity_mode':{'throughput_mode':'PAY_PER_REQUEST'}
} AND CLUSTERING ORDER BY("timestamp" DESC)
